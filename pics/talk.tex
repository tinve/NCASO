\begin{frame}
\frametitle{Update algorithm}

\begin{itemize}
\item Flip random spin
\item Find energy change $\Delta E$
\item Keep the spin flipped with probability $p = \frac{1}{1 + \exp{\frac{\Delta E}{T}}}$

\end{itemize}
